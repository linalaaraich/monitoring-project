New Services at a Glance

🤖 llm-triage (FastAPI :8090)

Receives Alertmanager webhook POSTs. Queries Prometheus, Loki, and Jaeger APIs to gather metric and log context around the alert window. Runs Drain3 log clustering in-process to identify anomalous log templates. Builds an enriched prompt and calls the local Ollama API. Returns the RCA as JSON; optionally triggers email via Alertmanager.

Python 3.11 FastAPI Drain3 httpx

🦙 Ollama (:11434)

Self-hosted LLM inference engine running inside Docker on monitoring-vm. Hosts llama3:8b (fallback: mistral:7b). Requires 8 GB RAM headroom on monitoring-vm. All inference stays on-premise — no external API calls, no data leaves the network.

main.py — Webhook Endpoint

roles/llm_triage/templates/main.py.j2

import logging, asyncio, json
from datetime import datetime, timezone
from fastapi import FastAPI, BackgroundTasks
from pydantic import BaseModel
from typing import List, Dict, Any

from context_fetcher import ContextFetcher
from drain_analyzer import DrainAnalyzer
from llm_client import LLMClient

# Structured logging — Promtail will pick this up
logging.basicConfig(
    level=logging.INFO,
    format='{"time":"%(asctime)s","level":"%(levelname)s","service":"llm-triage","msg":%(message)s}'
)
log = logging.getLogger("llm-triage")

app = FastAPI(title="LLM Triage Service", version="1.0.0")
fetcher = ContextFetcher()
analyzer = DrainAnalyzer()
llm = LLMClient()


class AlertPayload(BaseModel):
    receiver: str
    status: str
    alerts: List[Dict[str, Any]]
    groupLabels: Dict[str, str] = {}
    commonLabels: Dict[str, str] = {}
    commonAnnotations: Dict[str, str] = {}


@app.get("/health")
async def health():
    return {"status": "ok", "time": datetime.now(timezone.utc).isoformat()}


@app.post("/webhook")
async def webhook(payload: AlertPayload, background_tasks: BackgroundTasks):
    """Receive Alertmanager webhook, run RCA in background."""
    firing = [a for a in payload.alerts if a.get("status") == "firing"]
    if not firing:
        return {"status": "skipped", "reason": "no firing alerts"}

    background_tasks.add_task(run_rca, firing[0])
    return {"status": "accepted", "alert": firing[0].get("labels", {}).get("alertname")}


async def run_rca(alert: Dict[str, Any]):
    alertname = alert.get("labels", {}).get("alertname", "unknown")
    service   = alert.get("labels", {}).get("job", "spring")
    log.info(json.dumps({"event": "rca_start", "alert": alertname, "service": service}))

    try:
        context = await fetcher.gather(alert)
        anomalies = analyzer.find_anomalies(context["log_lines"])
        rca = await llm.generate_rca(alert, context, anomalies)

        result = {
            "alert": alertname, "service": service,
            "anomalous_templates": anomalies,
            "rca": rca,
            "generated_at": datetime.now(timezone.utc).isoformat(),
        }
        log.info(json.dumps({"event": "rca_complete", "result": result}))
    except Exception as exc:
        log.error(json.dumps({"event": "rca_error", "error": str(exc)}))
context_fetcher.py — Prometheus / Loki / Jaeger

roles/llm_triage/templates/context_fetcher.py.j2

import os, time, httpx
from typing import Dict, Any, List

PROMETHEUS_URL = os.getenv("PROMETHEUS_URL", "http://prometheus:9090")
LOKI_URL       = os.getenv("LOKI_URL", "http://loki:3100")
JAEGER_URL     = os.getenv("JAEGER_URL", "http://jaeger:16686")
WINDOW_MINUTES = int(os.getenv("CONTEXT_WINDOW_MINUTES", "15"))
LOKI_LIMIT     = int(os.getenv("LOKI_LIMIT", "200"))


class ContextFetcher:
    async def gather(self, alert: Dict[str, Any]) -> Dict[str, Any]:
        service    = alert.get("labels", {}).get("job", "spring")
        expr       = alert.get("labels", {}).get("prometheus_expr", "")
        end_ns     = int(time.time() * 1e9)
        start_ns   = end_ns - WINDOW_MINUTES * 60 * int(1e9)
        end_s      = int(time.time())
        start_s    = end_s - WINDOW_MINUTES * 60

        async with httpx.AsyncClient(timeout=20) as client:
            metrics, logs, traces = await asyncio.gather(
                self._fetch_metrics(client, expr, end_s),
                self._fetch_logs(client, service, start_ns, end_ns),
                self._fetch_traces(client, service, start_s, end_s),
                return_exceptions=True
            )

        return {
            "metrics_summary": metrics if not isinstance(metrics, Exception) else "unavailable",
            "log_lines":       logs    if not isinstance(logs,    Exception) else [],
            "traces_summary":  traces  if not isinstance(traces,  Exception) else "unavailable",
        }

    async def _fetch_metrics(self, client, expr, end_s) -> str:
        if not expr:
            return "no expression provided"
        r = await client.get(
            f"{PROMETHEUS_URL}/api/v1/query",
            params={"query": expr, "time": end_s}
        )
        r.raise_for_status()
        data = r.json().get("data", {}).get("result", [])
        return str(data[:5]) if data else "no data"

    async def _fetch_logs(self, client, service, start_ns, end_ns) -> List[str]:
        r = await client.get(
            f"{LOKI_URL}/loki/api/v1/query_range",
            params={
                "query": f'{{job="{service}"}}',
                "start": start_ns, "end": end_ns,
                "limit": LOKI_LIMIT, "direction": "backward"
            }
        )
        r.raise_for_status()
        streams = r.json().get("data", {}).get("result", [])
        lines = []
        for stream in streams:
            for ts, line in stream.get("values", []):
                lines.append(line)
        return lines

    async def _fetch_traces(self, client, service, start_s, end_s) -> str:
        r = await client.get(
            f"{JAEGER_URL}/api/traces",
            params={
                "service": service,
                "start": start_s * 1_000_000,
                "end": end_s * 1_000_000,
                "limit": 10
            }
        )
        r.raise_for_status()
        traces = r.json().get("data", [])
        if not traces:
            return "no recent traces"
        summary = []
        for t in traces[:5]:
            spans = t.get("spans", [])
            root  = spans[0] if spans else {}
            dur_ms = root.get("duration", 0) / 1000
            summary.append(f"{root.get('operationName','?')} — {dur_ms:.1f}ms — {len(spans)} spans")
        return "\n".join(summary)
drain_analyzer.py — Drain3 Log Clustering

roles/llm_triage/templates/drain_analyzer.py.j2

from drain3 import TemplateMiner
from drain3.template_miner_config import TemplateMinerConfig
from typing import List, Dict
from collections import Counter


class DrainAnalyzer:
    """
    Uses Drain3 online log template mining to cluster raw log lines.
    Anomalous templates = new/rare clusters not seen before (count == 1).
    """

    def find_anomalies(self, log_lines: List[str]) -> List[Dict]:
        if not log_lines:
            return []

        cfg = TemplateMinerConfig()
        cfg.load_defaults()
        cfg.parametrize_numeric_tokens = True   # replace numbers with <NUM>

        # Fresh miner per invocation — stateless by design
        miner = TemplateMiner(config=cfg)
        cluster_counts: Counter = Counter()

        for line in log_lines:
            result = miner.add_log_message(line.strip())
            if result and result.get("cluster_id"):
                cluster_counts[result["cluster_id"]] += 1

        # Anomalous: templates with very low frequency (count <= 2)
        anomalous = []
        for cluster in miner.drain.clusters:
            count = cluster_counts.get(cluster.cluster_id, 0)
            if count <= 2 and cluster.get_template():
                # Also flag ERROR/WARN templates regardless of frequency
                template = cluster.get_template()
                is_error = any(kw in template.upper() for kw in ["ERROR", "EXCEPTION", "FAIL", "WARN"])
                anomalous.append({
                    "template": template,
                    "count": count,
                    "is_error_pattern": is_error,
                })

        # Also include high-frequency ERROR templates (known but frequent = escalating)
        for cluster in miner.drain.clusters:
            template = cluster.get_template()
            count = cluster_counts.get(cluster.cluster_id, 0)
            if count > 10 and any(kw in template.upper() for kw in ["ERROR", "EXCEPTION"]):
                anomalous.append({
                    "template": template,
                    "count": count,
                    "is_error_pattern": True,
                    "note": "high-frequency error",
                })

        return anomalous[:10]  # cap at 10 templates to keep prompt size manageable
llm_client.py — Ollama API + Prompt Template

roles/llm_triage/templates/llm_client.py.j2

import os, json, httpx
from typing import Dict, Any, List

OLLAMA_URL   = os.getenv("OLLAMA_URL", "http://ollama:11434")
OLLAMA_MODEL = os.getenv("OLLAMA_MODEL", "llama3:8b")
LLM_TIMEOUT  = int(os.getenv("LLM_TIMEOUT", "300"))


class LLMClient:
    async def generate_rca(
        self,
        alert:     Dict[str, Any],
        context:   Dict[str, Any],
        anomalies: List[Dict]
    ) -> str:
        prompt = self._build_prompt(alert, context, anomalies)
        async with httpx.AsyncClient(timeout=LLM_TIMEOUT) as client:
            r = await client.post(
                f"{OLLAMA_URL}/api/generate",
                json={"model": OLLAMA_MODEL, "prompt": prompt, "stream": False}
            )
            r.raise_for_status()
        return r.json().get("response", "(empty response)").strip()

    def _build_prompt(
        self,
        alert:     Dict[str, Any],
        context:   Dict[str, Any],
        anomalies: List[Dict]
    ) -> str:
        alertname = alert.get("labels", {}).get("alertname", "unknown")
        severity  = alert.get("labels", {}).get("severity", "unknown")
        summary   = alert.get("annotations", {}).get("summary", "no summary")
        service   = alert.get("labels", {}).get("job", "unknown")

        anomaly_block = "None detected."
        if anomalies:
            lines = []
            for a in anomalies:
                flag = "[ERROR]" if a.get("is_error_pattern") else "[RARE]"
                lines.append(f"  {flag} {a['template']}  (seen {a['count']}x)")
            anomaly_block = "\n".join(lines)

        return f"""You are an expert site reliability engineer performing root cause analysis.

## ALERT
- Name: {alertname}
- Severity: {severity}
- Summary: {summary}
- Affected service: {service}

## METRICS CONTEXT (last 15 min)
{context.get("metrics_summary", "unavailable")}

## ANOMALOUS LOG PATTERNS (Drain3 analysis)
{anomaly_block}

## RECENT TRACES SUMMARY
{context.get("traces_summary", "unavailable")}

## TASK

How It Works

Log-based RCA event tracking — no extra infrastructure needed

The FastAPI service logs structured JSON to stdout. Docker captures it. Promtail (already running on monitoring-vm) ships it to Loki. Grafana can then query Loki for rca_complete events and display them as a table or stat panel.

Loki Retention for RCA Logs

Keep RCA logs longer than regular application logs

RCA outputs are valuable for post-incident review. Consider a longer retention period for the llm-triage job. Add a stream selector rule to the Loki config:

# In roles/loki/templates/loki.yml.j2 — per-stream retention (Loki 2.9+)
compactor:
  working_directory: /loki/compactor
  retention_enabled: true

limits_config:
  retention_period: 720h   # 30 days default
  per_stream_rate_limit: 3MB

# Per-stream retention override for RCA logs — 90 days
ruler:
  storage:
    type: local
